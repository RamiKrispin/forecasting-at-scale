featurs_df <- lapply(1:nrow(keys), function(i){
    d <- NULL
    d <- ts_obj |>
    dplyr::filter(area_name == keys$area_name[i],
                  process == keys$process[i])

    f <- tsfeatures::tsfeatures(d$value)
    f$area_name <-  keys$area_name[i]
    f$process <-  keys$process[i]
    f$nperiods <- NULL
    f$frequency <- NULL
    f$seasonal_period <- NULL
    return(f)
}) |>
